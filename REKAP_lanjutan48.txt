.hero-section {
    background: linear-gradient(135deg, #0d6efd 0%, #0a58ca 100%);

}

}

    border-radius: 50px;
    font-weight: 600;
}

    font-weight: 700;
}

.lead {
    font-weight: 300;
}

/* Responsive */

        text-align: center;
